window.toggleAdminMode = function() { if(document.getElementById('adminSection').classList.contains('hidden')) { document.getElementById('userSection').classList.add('hidden'); document.getElementById('loginSection').classList.remove('hidden'); } else { // 관리자 화면에서 다시 누르면 새로고침(사용자 화면 복귀) location.reload(); } }; // 로그인 체크 window.checkLogin = function() { if(document.getElementById('pwd').value === '1234') { document.getElementById('loginSection').classList.add('hidden'); document.getElementById('adminSection').classList.remove('hidden'); document.getElementById('title').innerText = "관리자 전용 관제실"; } else { alert("암호가 올바르지 않습니다."); } }; // 로그인 취소 window.cancelLogin = function() { document.getElementById('loginSection').classList.add('hidden'); document.getElementById('userSection').classList.remove('hidden'); }; // 배차 신청 window.submitDispatch = function() { const carIdx = document.getElementById('carSelect').value; const name = document.getElementById('userName').value; if(!name) return alert('이름을 입력하세요.'); if(!carIdx) return alert('차량을 선택하세요.'); cars[carIdx].status = false; alert(name + '님의 배차 신청이 완료되었습니다.'); updateUI(); }; // 초기 화면 그리기 updateUI();
